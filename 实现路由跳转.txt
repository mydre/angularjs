<script>
    var m = angular.module('hd', ['ui.router']);//继承了ui.router,那么其中的东西就都可以使用了
    m.config(['$stateProvider','$urlRouterProvider',function($stateProvider,$urlRouterProvider){
        $urlRouterProvider.otherwise('');
        $stateProvider.
            state('default',{
                url:'',
                template:'<h1>默认的页面</h1>'
            })
            .state('home',{
                url:'/home',
                templateUrl: 'view2/xx.html',
                // controller:['$scope',function($scope){
                //     $scope.name = '向军老师';
                // }]
                controller:'ctrl'
            })
            .state('lists',{
                url:'/lists',
                template:'<h1>列表页</h1>'
            })
    }]);
    m.controller('ctrl',['$state','$scope',function($state,$scope){
        $scope.data = [
            {title:'后盾网'},
            {title:'人人做后盾'}
        ];
        $scope.go=function(url){
            $state.go(url);
        }
    }]);
</script>
